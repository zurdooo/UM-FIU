# HW2 Programs

## Run the prebuilt Cpp executables, no need to recompile

./hw2_array

./hw2_string

## Compile and run the Cpp code

### hw2_arrays.cpp

g++ -std=c++23 hw2_arrays.cpp -o hw2_arrays ; ./hw2_arrays

### hw2_string.cpp

g++ -std=c++23 hw2_string.cpp -o hw2_string_cpp ; ./hw2_string_cpp

## Run the Python code

### hw2_string.py

python3 hw2_string.py

## Run the Ruby code

### hw2_string.rb

ruby hw2_string.rb
